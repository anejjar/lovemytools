## рџ—„пёЏ Database Schema (PostgreSQL via Drizzle)

```typescript
// src/db/schema.ts

// tools вЂ” mirrors the _registry but persists analytics
tools {
  id          uuid PK
  slug        text UNIQUE NOT NULL       -- "json-formatter"
  name        text NOT NULL              -- "JSON Formatter"
  category    text NOT NULL              -- "json"
  created_at  timestamp
}

// tool_usage вЂ” track how often each tool is used
tool_usage {
  id          uuid PK
  tool_id     uuid FK в†’ tools.id
  session_id  text                       -- anonymous session
  used_at     timestamp
}

// tool_feedback вЂ” thumbs up/down per tool
tool_feedback {
  id          uuid PK
  tool_id     uuid FK в†’ tools.id
  rating      smallint                   -- 1 (рџ‘Ќ) or -1 (рџ‘Ћ)
  message     text                       -- optional comment
  created_at  timestamp
}

// newsletter_subscribers
subscribers {
  id          uuid PK
  email       text UNIQUE NOT NULL
  subscribed_at timestamp
  source      text                       -- "home", "tool-page", "extension"
}

// page_views вЂ” lightweight SEO analytics
page_views {
  id          uuid PK
  path        text NOT NULL              -- "/tools/json-formatter"
  referrer    text
  country     text                       -- from Vercel edge headers
  viewed_at   timestamp
}
```
